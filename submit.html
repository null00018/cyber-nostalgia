--- .github/workflows/post.yml --- name: Create post from issue on: issues: types: [opened] jobs: build: runs-on: ubuntu-latest permissions: contents: write steps: - name: Checkout repo uses: actions/checkout@v3 - name: Extract issue content uses: actions/github-script@v7 id: issue with: script: | const body = context.payload.issue.body; const fs = require('fs'); fs.writeFileSync('issue.md', body); - name: Parse front matter id: vars run: | TITLE=$(grep '^title:' issue.md | sed 's/title: //') DATE=$(grep '^date:' issue.md | sed 's/date: //') if [ -z "$DATE" ]; then DATE=$(date +"%Y%m%d") fi BODY=$(awk '/^---/{f++} f==2' issue.md | tail -n +2) echo "title=$TITLE" >> $GITHUB_OUTPUT echo "date=$DATE" >> $GITHUB_OUTPUT echo "$BODY" > content.txt - name: Create HTML file run: | FILENAME=post$(date +%s).html echo "FILENAME=$FILENAME" >> $GITHUB_ENV cat < $FILENAME ${{ steps.vars.outputs.title }}

＞＞ ${{ steps.vars.outputs.title }}

${{ steps.vars.outputs.date }}

$(cat content.txt | sed 's/^/ /')

←START

EOF - name: Commit post run: | git config user.name "github-actions" git config user.email "github-actions@github.com" git add $FILENAME git commit -m "Add $FILENAME" git push - name: Trigger index update run: | echo "Post created: $FILENAME" # あなたの既存の index.html 自動更新ワークフローがここで動作する想定です
